36 auto ReceiverModule::run() -> void {
37  std::vector<unsigned short> buf(numberOfProjections_*16 + sizeof(std::size_t)/sizeof(unsigned short));
38  unsigned int sinoSize = numberOfProjections_*16;
39  std::size_t headerSize{sizeof(std::size_t)/sizeof(unsigned short)};
40  int numBytes = 0;
41  int timeoutS = 30;
42  while(true){
43  numBytes = udpServer_.timed_recv((char*)(buf.data()), buf.size()*sizeof(unsigned short), timeoutS);
44  if(numBytes < 0) break;
45  std::size_t index = *((std::size_t *)buf.data());
46  int diff = index - lastIndex_ - 1;
47  if(diff > 0 && index > 0){
48  BOOST_LOG_TRIVIAL(warning) << "ReceiverModule " << moduleID_ << ": Lost package or wrong order. Last " << lastIndex_ << " new: " << index;
49  lastIndex_ = index;
50  continue;
51  }
52  lastIndex_ = index;
53  BOOST_LOG_TRIVIAL(debug) << "ReceiverModule " << moduleID_ << " received packet " << index;
54  auto it = std::count(buf.cbegin()+headerSize, buf.cend(), 0);
55  std::copy(buf.cbegin() + headerSize, buf.cend(), buffer_.begin() + sinoSize * (index%bufferSize_));
56  notification_.notify(moduleID_, index);
57  }
58  BOOST_LOG_TRIVIAL(info) << "ReceiverModul " << moduleID_ << ": No packets arriving since " << timeoutS << "s. Finishing.";
59 }
