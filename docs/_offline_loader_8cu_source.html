42 auto OfflineLoader::loadImage() -> ddrf::Image<manager_type> {
43  if (index_ >= numberOfFrames_ * numberOfPlanes_)
44  return ddrf::Image<manager_type>();
45  BOOST_LOG_TRIVIAL(debug) << "risa::Offlineloader: Loading image with index " << index_;
46  auto sino = ddrf::MemoryPool<manager_type>::instance()->requestMemory(memoryPoolIndex_);
47 #pragma omp parallel for
48  for (auto detModInd = 0; detModInd < numberOfDetectorModules_; detModInd++) {
49  ifstreams_[detModInd].get()->seekg(index_ * numberOfProjections_ * numberOfDetectorsPerModule_ * sizeof(unsigned short),
50  std::ios::beg);
51  ifstreams_[detModInd].get()->read((char*) sino.container().get()+ detModInd * numberOfDetectorsPerModule_
52  * sizeof(unsigned short) * numberOfProjections_, numberOfProjections_ * numberOfDetectorsPerModule_ * sizeof(unsigned short));
53  }
54 // ifstreams_[0].get()->seekg(index_ * numberOfProjections_ * numberOfDetectors_ * sizeof(float),
55 // std::ios::beg);
56 // ifstreams_[0].get()->read(
57 // (char*) sino.container().get(), numberOfProjections_ * numberOfDetectors_ * sizeof(float));
58  BOOST_LOG_TRIVIAL(debug)<< "Loading sinogram " << index_;
59  sino.setIdx(index_);
60  sino.setPlane(index_%numberOfPlanes_);
61  index_++;
62  sino.setStart(std::chrono::high_resolution_clock::now());
