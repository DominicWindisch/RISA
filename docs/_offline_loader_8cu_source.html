45 auto OfflineLoader::loadImage() -> ddrf::Image<manager_type> {
46  if (buffer_.empty())

53  bestCaseTime_ = duration;
54  if (duration > worstCaseTime_)
55  worstCaseTime_ = duration;
56  }
57  buffer_.pop();
58  if (index_ < stopFrame_) {
59  auto img = ddrf::MemoryPool<manager_type>::instance()->requestMemory(
60  memoryPoolIndex_);
61  img.setIdx(index_);
62  buffer_.push(std::move(img));

64  }
65  tmr_.start();
66  sino.setStart(std::chrono::high_resolution_clock::now());

145 auto OfflineLoader::readConfig(const std::string& configFile) -> bool {

148  if (configReader.lookupValue("numberOfFanDetectors", numberOfDetectors_)

150  numberOfDetectorModules_)

153  && configReader.lookupValue("dataFileEnding", fileEnding_)
154  && configReader.lookupValue("numberOfPlanes", numberOfPlanes_)
155  && configReader.lookupValue("samplingRate", samplingRate)
156  && configReader.lookupValue("scanRate", scanRate)
157  && configReader.lookupValue("numberOfDataFrames", numberOfFrames_)) {
158  numberOfProjections_ = samplingRate * 1000000 / scanRate;
