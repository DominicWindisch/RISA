69 auto Masking::process(input_type&& img) -> void {
70  if (img.valid()) {

94 auto Masking::processor(const int deviceID) -> void {
95  //nvtxNameOsThreadA(pthread_self(), "CropImage");
96  CHECK(cudaSetDevice(deviceID));
97  dim3 blocks(16, 16);
98  dim3 grids(std::ceil(numberOfPixels_ / 16.0),
99  std::ceil(numberOfPixels_ / 16.0));
100  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::CropImage: Running Thread for Device " << deviceID;
101  while (true) {
102  auto img = imgs_[deviceID].take();
103  if (!img.valid())
104  break;
105  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: CropImageing image with Index " << img.index();
106 
107  mask<<<grids, blocks, 0, streams_[deviceID]>>>(img.container().get(),
108  0.0 ,numberOfPixels_);
109  CHECK(cudaPeekAtLastError());
110 
111  //wait until work on device is finished
112  CHECK(cudaStreamSynchronize(streams_[deviceID]));
113  results_.push(std::move(img));
114 
115  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::CropImage: CropImageing image with Index " << img.index() << " finished.";
116  }
117 }

128 __global__ void mask(float* __restrict__ img, const float value, const int numberOfPixels) {
129  const auto x = ddrf::cuda::getX();
130  const auto y = ddrf::cuda::getY();
131  if (x >= numberOfPixels || y >= numberOfPixels)
132  return;
133  const float center = (numberOfPixels - 1.0) * 0.5;
134  const float dX = x - center;
135  const float dY = y - center;
136  const float distance = dX * dX + dY * dY;
137  if (distance > numberOfPixels * numberOfPixels * 0.25) {
138  img[x + numberOfPixels * y] = value;
139  }
140 }
