79 auto Attenuation::process(input_type&& sinogram) -> void {
80  if (sinogram.valid()) {
81  BOOST_LOG_TRIVIAL(debug)<< "Attenuation: Image arrived with Index: " << sinogram.index() << "to device " << sinogram.device();
82  sinograms_[sinogram.device()].push(std::move(sinogram));
83  } else {
84  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Received sentinel, finishing.";
85 
86  //send sentinal to processor thread and wait 'til it's finished
87  for(auto i = 0; i < numberOfDevices_; i++) {
88  sinograms_[i].push(input_type());
89  }
90 
91  for(auto i = 0; i < numberOfDevices_; i++) {
92  processorThreads_[i].join();
93  }
94  //push sentinel to results for next stage
95  results_.push(output_type());
96  BOOST_LOG_TRIVIAL(info) << "recoLib::cuda::Attenuation: Finished.";
97  }
98 }

104 auto Attenuation::processor(const int deviceID) -> void {
105  //nvtxNameOsThreadA(pthread_self(), "Attenuation");
106  CHECK(cudaSetDevice(deviceID));
107  auto avgDark_d = ddrf::cuda::make_device_ptr<float>(avgDark_.size());
108  auto avgReference_d = ddrf::cuda::make_device_ptr<float>(
109  avgReference_.size());
110  auto mask_d = ddrf::cuda::make_device_ptr<float>(
111  numberOfDetectors_ * numberOfProjections_);
112  CHECK(
113  cudaMemcpyAsync(avgDark_d.get(), avgDark_.data(),
114  sizeof(float) * avgDark_.size(), cudaMemcpyHostToDevice,
115  streams_[deviceID]));
116  CHECK(
117  cudaMemcpyAsync(avgReference_d.get(), avgReference_.data(),
118  sizeof(float) * avgReference_.size(), cudaMemcpyHostToDevice,
119  streams_[deviceID]));
120  //compute mask for relevant area
121  std::vector<float> mask;
122  relevantAreaMask(mask);
123  CHECK(
124  cudaMemcpyAsync(mask_d.get(), mask.data(), sizeof(float) * mask.size(),
125  cudaMemcpyHostToDevice, streams_[deviceID]));
126 
127  dim3 blocks(blockSize2D_, blockSize2D_);
128  dim3 grids(std::ceil(numberOfDetectors_ / (float)blockSize2D_),
129  std::ceil(numberOfProjections_ / (float)blockSize2D_));
130  float temp = pow(10, -5);
131  CHECK(cudaStreamSynchronize(streams_[deviceID]));
132  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Attenuation: Running Thread for Device " << deviceID;
133 
134  while (true) {
135  auto sinogram = sinograms_[deviceID].take();
136  if (!sinogram.valid())
137  break;
138  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Attenuationing image with Index " << sinogram.index();
139 
140  auto sino =
141  ddrf::MemoryPool<deviceManagerType>::instance()->requestMemory(
142  memoryPoolIdxs_[deviceID]);
143 
144  computeAttenuation<<<grids, blocks, 0, streams_[deviceID]>>>(
145  sinogram.container().get(), mask_d.get(), sino.container().get(),
146  avgReference_d.get(), avgDark_d.get(), temp, numberOfDetectors_,
147  numberOfProjections_, sinogram.plane());
148  CHECK(cudaPeekAtLastError());
149 
150  sino.setIdx(sinogram.index());
151  sino.setDevice(deviceID);
152  sino.setPlane(sinogram.plane());
153  sino.setStart(sinogram.start());
154 
155  //wait until work on device is finished
156  CHECK(cudaStreamSynchronize(streams_[deviceID]));
157  results_.push(std::move(sino));
158 
159  BOOST_LOG_TRIVIAL(debug)<< "recoLib::cuda::Attenuation: Attenuationing image with Index " << sinogram.index() << " finished.";
160  }

163 auto Attenuation::init() -> void {
164  //create filter function
165  std::vector<double> filterFunction{0.5, 1.0, 1.0, 1.0, 1.5, 2.0, 3.0, 3.5, 2.0, 3.5, 3.0, 2.0, 1.5, 1.0, 1.0, 1.0, 0.5};
166  double sum = std::accumulate(filterFunction.cbegin(), filterFunction.cend(), 0.0);
167  std::transform(filterFunction.begin(), filterFunction.end(), filterFunction.begin(),
168  std::bind1st(std::multiplies<double>(), 1.0/sum));
169 
170  //read and average reference input values
171  std::vector<unsigned short> referenceValues;
172  if(pathReference_.back() != '/')
173  pathReference_.append("/");
174  std::string refPath = pathReference_ + "ref_empty_tomograph_repaired_DetModNr_";
175  readInput(refPath, referenceValues, numberOfRefFrames_);
176  //interpolate reference measurement
177  for(auto i = 0; i < numberOfRefFrames_*numberOfPlanes_; i++){
178  std::vector<int> defectDetectors(numberOfProjections_*numberOfDetectors_);
179  findDefectDetectors(referenceValues.data()+i*numberOfDetectors_*numberOfProjections_, filterFunction, defectDetectors, numberOfDetectors_, numberOfProjections_,
180  threshMin_, threshMax_);
181  interpolateDefectDetectors(referenceValues.data()+i*numberOfDetectors_*numberOfProjections_, defectDetectors, numberOfDetectors_, numberOfProjections_);
182  }
183  computeAverage(referenceValues, avgReference_);
184 
185  //read and average dark input values
186  std::vector<unsigned short> darkValues;
187  if(pathDark_.back() != '/')
188  pathDark_.append("/");
189  std::string darkPath = pathDark_ + "dark_192.168.100_DetModNr_";
190  readInput(darkPath, darkValues, numberOfDarkFrames_);
191  computeDarkAverage(darkValues, avgDark_);
192  //interpolate dark average
193  for(auto j = 0; j < numberOfPlanes_; j++){
194  for(auto i = 0; i < numberOfDetectors_; i++){
195  if(avgDark_[i + j * numberOfDetectors_] > 300.0){
196  BOOST_LOG_TRIVIAL(info) << "Interpolating dark value at detector " << i << " in plane " << j;
197  avgDark_[numberOfDetectors_ * j + i] =
198  0.5 * (avgDark_[numberOfDetectors_ * j + (i + 1)%numberOfDetectors_] +
199  avgDark_[numberOfDetectors_ * j + (i - 1)%numberOfDetectors_]);
200  }
201  }
202  }

206 auto Attenuation::computeDarkAverage(const std::vector<T>& values, std::vector<float>& average) -> void {
207  average.resize(numberOfDetectors_*numberOfPlanes_, 0.0);
208  float factor = 1.0/ (float)((float)numberOfDarkFrames_*(float)numberOfProjections_);
209  factor = 0.0;
210  for(auto i = 0; i < numberOfDarkFrames_; i++){
211  for(auto planeInd = 0; planeInd < numberOfPlanes_; planeInd++){
212  for(auto detInd = 0; detInd < numberOfDetectors_; detInd++){
213  for(auto projInd = 0; projInd < numberOfProjections_; projInd++){
214  const float val = (float)values[detInd + numberOfDetectors_*projInd + (i*numberOfPlanes_+planeInd)*numberOfDetectors_*numberOfProjections_];
215  average[detInd + planeInd*numberOfDetectors_] += val * factor;
216  }
217  }
218  }

268  tmr2.start();

271  std::vector<T> content;

278  //allocate memory in vector

287  tmr2.stop();

289  values.resize(fileContents[0].size() * numberOfDetectorModules_);
290  for (auto i = 0; i < numberOfFrames; i++) {
291  for (auto planeInd = 0; planeInd < numberOfPlanes_; planeInd++) {
292  for (auto projInd = 0; projInd < numberOfProjections_; projInd++) {
293  for (auto detModInd = 0; detModInd < numberOfDetectorModules_;
294  detModInd++) {
295  unsigned int startIndex = projInd * numberOfDetPerModule
296  + (planeInd + i * numberOfPlanes_) * numberOfDetPerModule * numberOfProjections_;
297  unsigned int indexSorted = detModInd * numberOfDetPerModule
298  + projInd * numberOfDetectors_
299  + (planeInd + i * numberOfPlanes_) * numberOfDetectors_ * numberOfProjections_;
300  std::copy(fileContents[detModInd].begin() + startIndex,
301  fileContents[detModInd].begin() + startIndex
302  + numberOfDetPerModule,
303  values.begin() + indexSorted);
304  }
305  }
306  }
307  }
308  tmr1.stop();
309  double totalFileSize = numberOfProjections_*numberOfDetectors_*numberOfPlanes_*numberOfRefFrames_*sizeof(unsigned short)/1024.0/1024.0;
310  BOOST_LOG_TRIVIAL(info)<< "recoLib::cuda::Attenuation: Reading and sorting reference input took " << tmr1.elapsed() << " s, " << totalFileSize/tmr2.elapsed() << " MByte/s.";

313 template<typename T>
314 auto Attenuation::relevantAreaMask(std::vector<T>& mask) -> void {
315  unsigned int ya, yb, yc, yd, ye;
316  unsigned int yMin, yMax;
317  double lowerLimit = (lowerLimOffset_ + sourceOffset_) / 360.0;
318  double upperLimit = (upperLimOffset_ + sourceOffset_) / 360.0;
319  //fill whole mask with ones and mask out the unrelevant parts afterwards
320  mask.resize(numberOfProjections_ * numberOfDetectors_);
321  std::fill(mask.begin(), mask.end(), 1.0);
322 
323  ya = std::round(lowerLimit * numberOfProjections_);
324  yb = ya;
325  yc = std::round(upperLimit * numberOfProjections_);
326  yd = yc;
327 
328  //slope of the straight
329  double m = ((double)ya - (double)yd) / ((double)xa_ - (double)xd_);
330 
331  ye = std::round((double)yc + ((double)xe_ - (double)xc_) * m);
332 
333  for (unsigned int x = 0; x <= xa_; x++) {
334  yMin = ya;
335  yMax = std::round(ye + m * x);
336  for (auto y = yMin; y < yMax; y++)
337  mask[x + y * numberOfDetectors_] = 0.0;
338  }
339 
340  for (auto x = xa_; x <= xc_; x++) {
341  yMin = std::round(ya + m * (x - xa_));
342  yMax = std::round(ye + m * x);
343  for (auto y = yMin; y < yMax; y++)
344  mask[x + y * numberOfDetectors_] = 0.0;
345  }
346 
347  for (auto x = xc_; x <= xd_; x++) {
348  yMin = std::round(ya + m * (x - xa_));
349  yMax = yd;
350  for (auto y = yMin; y < yMax; y++)
351  mask[x + y * numberOfDetectors_] = 0.0;
352  }
353 
354  for (auto x = xb_; x <= xf_; x++) {
355  yMin = yb;
356  yMax = std::round(yb + m * (x - xb_));
357  for (auto y = yMin; y < yMax; y++)
358  mask[x + y * numberOfDetectors_] = 0.0;
359  }
360 
361  std::fill(mask.begin(),
362  mask.begin() + lowerLimit * numberOfDetectors_ * numberOfProjections_,
363  0.0);
364  std::fill(
365  mask.begin() + upperLimit * numberOfProjections_ * numberOfDetectors_,
366  mask.end(), 0.0);
367 }

418 __global__ void computeAttenuation(
419  const unsigned short* __restrict__ sinogram_in,
420  const float* __restrict__ mask, float* __restrict__ sinogram_out,
421  const float* __restrict__ avgReference, const float* __restrict__ avgDark,
422  const float temp, const int numberOfDetectors,
423  const int numberOfProjections, const int planeId) {
424 
425  auto x = ddrf::cuda::getX();
426  auto y = ddrf::cuda::getY();
427  if (x >= numberOfDetectors || y >= numberOfProjections)
428  return;
429 
430  auto sinoIndex = numberOfDetectors * y + x;
431 
432  float numerator = (float) (sinogram_in[sinoIndex])
433  - avgDark[planeId * numberOfDetectors + x];
434 
435  float denominator = avgReference[planeId * numberOfDetectors * numberOfProjections + sinoIndex]
436  - avgDark[planeId * numberOfDetectors + x];
437 
438  if (numerator < temp)
439  numerator = temp;
440  if (denominator < temp)
441  denominator = temp;
442 
443  //comutes the attenuation and multiplies with mask for hiding the unrelevant region
444  sinogram_out[sinoIndex] = -log(numerator / denominator) * mask[sinoIndex];
445 
446 }
